#include "UWBFeature.h"

UWBFeature uwbFeature;

void setup()
{
    Serial.begin(115200);
    while (!Serial) delay(10);

    Serial.println("UWB ONLY TEST");
    uwbFeature.begin();
}

void loop()
{
    uwbFeature.update();

    Serial.print("D1=");
    if (uwbFeature.hasValidDistance())
        Serial.print(uwbFeature.getDistanceCm());
    else
        Serial.print("--");

    Serial.println(" cm");
    delay(100);
}
